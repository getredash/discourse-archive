Extended Dockerfile:

FROM redash/redash:latest

USER root

# Oracle instantclient

RUN unzip /tmp/instantclient-sqlplus-linux.x64-12.2.0.1.0.zip -d /usr/local/

RUN apt-get clean -y

ENV REDASH_ADDITIONAL_QUERY_RUNNERS=redash.query_runner.oracle
